create table log_change_Takes(
    Time_Of_Change timestamp(2),
    ID varchar(5),
    course_id varchar(8),
    sec_id varchar(8),
    semester varchar(6),
    year numeric(4, 0),
    grade varchar(2)
);

create or replace trigger log_takes
after update on takes
for each row
begin
insert into log_change_Takes values(current_timestamp, :OLD.ID, :OLD.course_id, :OLD.sec_id, :OLD.semester, :OLD.year, :OLD.grade);
end;

create table Old_Data_Instructor(
    ID varchar(5),
    name varchar(20) not null,
    dept_name varchar(20),
    salary numeric(8, 2) check(salary>29000),
    primary key(ID),
    foreign key(dept_name) references Department on delete set null
);

create or replace trigger log_instructor
after update on instructor
for each row
begin
insert into Old_Data_Instructor values(:OLD.ID, :OLD.name, :OLD.dept_name, :OLD.salary);
end;
