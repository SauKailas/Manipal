create or replace procedure departments (dname in instructor.dept_name%TYPE) is
cursor c1 is select title from course where dept_name=dname;
cursor c2 is select name from instructor where dept_name=dname;
begin
	dbms_output.put_line('Instructors in ' || dname || ' are: ');
	for s in c2
	loop 	
		dbms_output.put_line(s.name);
	end loop;
	
	dbms_output.put_line('Courses Offered by ' || dname || ' are: ');
	for s in c1
	loop 
		dbms_output.put_line(s.title);
	end loop;
end;
/
    
declare
    dname instructor.dept_name%TYPE;
begin
	dname:='&Department_Name';
	departments(dname);
end;
/

create or replace procedure course_popular is
cursor c1 is (select title, max(no_stu) as count_stu
				from (select title, count(distinct ID) as no_stu
						from takes natural join course
    					group by title));
begin
	for c in c1
	loop
    	dbms_output.put_line('Most Popular Department is ' || c.title || ' with ' || c.count_stu || ' students. ');
	end loop;
end;
/

declare
begin
	course_popular;
end;
/

create or replace function square(n integer)
	return integer as
    n_squared integer;
	begin
		n_squared:=n*n;
		return n_squared;
	end;
/

declare
    num integer;
	num_squared integer;
begin
    num:='&Number';
	dbms_output.put_line('Square of ' || num || ' is ' || square(num));
end;
